        // ─── GFR Calculation (CKD-EPI) ──────────────
        [SqlFunction(Name = "fn_CLR_CalculateGFR")]
        public static SqlDouble CalculateGFR(
            SqlDouble creatinine, SqlInt32 age,
            SqlString gender, SqlString race)
        {
            double scr = creatinine.Value;
            double kappa  = (gender.Value == "F") ? 0.7   : 0.9;
            double alpha  = (gender.Value == "F") ? -0.329 : -0.411;
            double gFactor= (gender.Value == "F") ? 1.018 : 1.0;

            double gfr = 141
                * Math.Pow(Math.Min(scr / kappa, 1.0), alpha)
                * Math.Pow(Math.Max(scr / kappa, 1.0), -1.209)
                * Math.Pow(0.993, age.Value)
                * gFactor;

            return new SqlDouble(Math.Round(gfr, 1));
        }
