-- Contains CROSS-DATABASE QUERIES to PatientDB
-- ============================================
USE BillingDB;
GO

CREATE PROCEDURE dbo.usp_ProcessEncounterCharges
    @EncounterID INT
AS BEGIN
    SET NOCOUNT ON;
    DECLARE @PatientID INT;

    BEGIN TRY
        BEGIN TRANSACTION;

        -- ⚠ CROSS-DATABASE QUERY: BillingDB → PatientDB
        SELECT @PatientID = e.PatientID
        FROM PatientDB.dbo.Encounters e
        WHERE e.EncounterID = @EncounterID
          AND e.EncounterStatus = 'ACTIVE';

        -- ⚠ CROSS-DATABASE INSERT pulling from PatientDB
        INSERT INTO dbo.BillingCharges (
            EncounterID, PatientID, ChargeCode, CPTCode,
            ChargeDescription, ServiceDate, UnitPrice, ChargeAmount,
            DepartmentCode, PerformingPhysicianNPI, DiagnosisCode1
        )
        SELECT
            p.EncounterID,
            p.PatientID,
            cm.ChargeCode,
            p.CPTCode,
            p.CPTDescription,
            p.ProcedureDate,
            cm.StandardCharge,
            cm.StandardCharge,
            -- Cross-DB lookup: department code
            (SELECT TOP 1 d.DepartmentCode
             FROM PatientDB.dbo.Departments d
             WHERE d.DepartmentID = e.DepartmentID),
            -- Cross-DB lookup: physician NPI
            (SELECT TOP 1 ph.NPI
             FROM PatientDB.dbo.Physicians ph
             WHERE ph.PhysicianID = p.PerformingPhysicianID),
            -- Cross-DB lookup: primary diagnosis
            (SELECT TOP 1 dx.ICDCode
             FROM PatientDB.dbo.Diagnoses dx
             WHERE dx.EncounterID = @EncounterID
               AND dx.DiagnosisType = 'PRIMARY')
        FROM PatientDB.dbo.Procedures p
        INNER JOIN PatientDB.dbo.Encounters e
            ON p.EncounterID = e.EncounterID
        LEFT JOIN dbo.Chargemaster cm
            ON p.CPTCode = cm.CPTCode
        WHERE p.EncounterID = @EncounterID
          AND p.ProcedureStatus = 'COMPLETED';

        COMMIT TRANSACTION;
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        THROW;
    END CATCH
END;
