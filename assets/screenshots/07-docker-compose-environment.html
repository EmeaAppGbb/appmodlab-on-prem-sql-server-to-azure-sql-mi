docker-compose.yml 🐳 3 Containers 
version: '3.8'

services:

  sqlserver:                      # Main Lakeview instance
    image: mcr.microsoft.com/mssql/server:2019-latest
    container_name: lakeview-sqlserver
    environment:
      - ACCEPT_EULA=Y
      - SA_PASSWORD=LakeviewMedical2024!
      - MSSQL_PID=Developer
      - MSSQL_AGENT_ENABLED=true
    ports:
      - "1433:1433"
    volumes:
      - ./Deployment:/scripts
      - ./Databases:/databases
      - ./SeedData:/seeddata
    healthcheck:
      test: sqlcmd -S localhost -U sa -Q "SELECT 1"
      interval: 10s

  pharmacy-db:                   # Simulated pharmacy system
    image: mcr.microsoft.com/mssql/server:2019-latest
    container_name: pharmacy-system
    ports:
      - "1434:1433"

  insurance-db:                  # Simulated insurance system
    image: mcr.microsoft.com/mssql/server:2019-latest
    container_name: insurance-system
    ports:
      - "1435:1433"

networks:
  lakeview-network:
    driver: bridge
