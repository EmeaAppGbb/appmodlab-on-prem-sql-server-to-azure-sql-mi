── NIGHTLY BILLING BATCH ──────────────────────
EXEC sp_add_job @job_name = N'LMC - Nightly Billing Batch';
-- Schedule: Daily at 2:00 AM
-- Steps: Post room charges → Create claims → Generate invoices
-- Calls: BillingDB.dbo.usp_BatchNightlyBilling

── INSURANCE CLAIMS SUBMISSION ────────────────
EXEC sp_add_job @job_name = N'LMC - Daily Insurance Claims';
-- Schedule: Twice daily at 6:00 AM and 6:00 PM
-- Steps: Submit pending → Process responses → Flag denials
-- Uses: Linked server INSURANCE_CLEARINGHOUSE

── MONTHLY DATA ARCHIVAL ──────────────────────
EXEC sp_add_job @job_name = N'LMC - Monthly Data Archival';
-- Schedule: 1st Sunday of month at 1:00 AM
-- Steps: Archive old records → Rebuild indexes
-- Cursor-based batch processing (1000 rows/batch)

── STATISTICS UPDATE ──────────────────────────
EXEC sp_add_job @job_name = N'LMC - Statistics Update';
-- Schedule: Daily at 3:00 AM
-- FULLSCAN on weekends, SAMPLE 50% on weekdays

── DATABASE BACKUP ────────────────────────────
EXEC sp_add_job @job_name = N'LMC - Full Database Backup';
-- Schedule: Daily at 11:00 PM
-- Full backup with COMPRESSION + CHECKSUM
-- 30-day retention, network share \\BACKUP-SERVER\

EXEC sp_add_job @job_name = N'LMC - Transaction Log Backup';
-- Schedule: Every 15 minutes
-- FULL recovery databases only
